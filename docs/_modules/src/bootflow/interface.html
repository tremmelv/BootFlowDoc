from src.bootflow.name_registry import NameRegistry


[docs]class Interface:
    """
    A class to represent an interface with a name and transfer speed.

    Attributes:
    -----------
    name : str
        The name of the interface.
    transferSpeed : float
        The transfer speed of the interface.
    comment : str, optional
        Additional comments about the interface (default is an empty string).
    """

    def __init__(self, name: str, transferSpeed: float, comment= ""):
        """
        Initializes an Interface instance with a name, transfer speed, and optional comment.

        Parameters:
        -----------
        name : str
            The name of the interface.
        transferSpeed : float
            The transfer speed of the interface.
        comment : str, optional
            Additional comments about the interface (default is an empty string).

        Raises:
        -------
        ValueError
            If the name is empty.
        """
        if not name:
            raise ValueError("name cannot be empty")
        NameRegistry.register_name(name)
        self.comment = comment
        self.name = name
        self.transferSpeed = transferSpeed

    def __del__(self):
        """
        Destructor method to unregister the name when the instance is deleted.
        """
        self.cleanup()


[docs]    def cleanup(self):
        """
        Method to unregister the boot event's name from the NameRegistry.
        """
        NameRegistry.unregister_name(self.name)


    def __str__(self):
        """
        Returns a string representation of the Interface instance.

        Returns:
        --------
        str
            A string representation of the Interface instance.
        """
        return f"Interface(name={self.name}, transferSpeed={self.transferSpeed}, comment={self.comment})"
