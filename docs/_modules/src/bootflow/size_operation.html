[docs]class SizeOperation(SizeEvent):
    """
    Represents an operation in the boot flow that involves the processing of software 
    based on its size and a specified operation speed.

    Inherits from:
    --------------
    SizeEvent : The base class for events related to size-based operations.

    Attributes:
    -----------
    operationSpeed : float or str
        The speed at which the size operation is performed. Can be a direct value or a variable name.
    duration : float or str
        The calculated duration of the operation in milliseconds.
    """

    def __init__(self, name: str, software: Software, operationSpeed, processor, comment: str = ""):
        """
        Initializes a SizeOperation object.

        Parameters:
        -----------
        name : str
            The name of the size operation event.
        software : Software
            The software associated with the size operation.
        operationSpeed : float or str
            The speed at which the operation is performed. Can be a direct value or a variable name.
        processor :
            The processor associated with this event.
        comment : str, optional
            An optional comment or description of the event. Defaults to "".
        """
        super().__init__(name, software, processor, comment)
        self.operationSpeed = operationSpeed
        self.duration = self.calculate_duration()


[docs]    def calculate_duration(self) -> Union[float, str]:
        """
        Calculates the duration of the size operation based on the software size 
        and the operation speed.

        Returns:
        --------
        float or str :
            The duration of the operation in milliseconds, or "-" if the size or speed is unavailable or the software was deleted.
        """
        if self.software == None or isinstance(self.software.get_size(), str) or isinstance(self.get_operation_speed(), str):
            return "-"
        else:
            return (self.software.get_size() / self.get_operation_speed()) * 1e3

    

[docs]    def get_operation_speed(self) -> Union[float, str]:
        """
        Retrieves the operation speed. If the speed is defined as a variable, 
        it checks and retrieves the value from the processor's boot flow variables.

        Returns:
        --------
        float or str :
            The operation speed as a float, or "TBD" if the value is not yet determined.
        """
        if isinstance(self.operationSpeed, str):
            self.check_variable_syntax(self.operationSpeed)
            variable_name = self.operationSpeed.replace('$', '')
            if variable_name in self.processor.boot_flow.variables:
                value = self.processor.boot_flow.variables[variable_name]
                return value
            else:
                self.processor.boot_flow.variables[variable_name] = "TBD"
                return "TBD"
        else:
            return self.operationSpeed


    def __str__(self) -> str:
        """
        Returns a string representation of the SizeOperation object.

        Returns:
        --------
        str :
            A string that represents the SizeOperation event.
        """
        return f"SizeOperation(name={self.name}, software={self.software}, operationSpeed={self.operationSpeed}, processor={self.processor.name})"
    

[docs]    def to_dict(self) -> dict:
        """
        Converts the SizeOperation object into a dictionary format.

        Returns:
        --------
        dict :
            A dictionary representation of the SizeOperation event, including type, name, software, operation speed, comment, and group.
        """
        return {
            "type": "SizeOperation",
            "name": self.name,
            "software": self.software.to_dict(),
            "operationSpeed": self.operationSpeed,
            "comment": self.comment,
            "group": self.group.name if self.group else None
        }

    def from_dict(cls, data: dict, processor, boot_flow):
        """
        Creates a SizeOperation object from a dictionary representation.

        Parameters:
        -----------
        data : dict
            A dictionary containing the SizeOperation event data.
        processor :
            The processor associated with the SizeOperation event.
        boot_flow :
            The boot flow object containing available software and groups.

        Returns:
        --------
        SizeOperation :
            A new SizeOperation object initialized with the data from the dictionary.
        """
        software_name = data["software"]["name"]
        used_software = boot_flow.software[software_name]
        new_event = cls(data["name"], used_software, data["operationSpeed"], processor, data["comment"])
        if data["group"]:
            group = boot_flow.groups[data["group"]]
            group.add_to_group(new_event)
        return new_event
