[docs]    def add_to_group(self, event):
        """
        Adds an event to the group.

        Parameters:
        -----------
        event : BootEvent
            The event to be added to the group.
        """
        self.members[event.name] = event
        event.group = self

    

[docs]    def remove_from_group(self, event):
        """
        Removes an event from the group.

        Parameters:
        -----------
        event : BootEvent
            The event to be removed from the group.
        """
        if event.name in self.members:
            del self.members[event.name]
            event.group = None
