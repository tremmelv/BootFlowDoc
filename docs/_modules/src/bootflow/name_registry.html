[docs]class NameRegistry:
    """
    A registry to manage unique names.

    Attributes:
    -----------
    _names : set
        A set to store unique names.
    """

    _names = set()


[docs]    @classmethod
    def register_name(cls, name: str):
        """
        Registers a name in the registry. Raises an error if the name is already in use.

        Parameters:
        -----------
        name : str
            The name to be registered.

        Raises:
        -------
        ValueError
            If the name is already in use.
        """
        if name in cls._names:
            raise ValueError(f"Name '{name}' is already in use.")
        cls._names.add(name)



[docs]    @classmethod
    def unregister_name(cls, name: str):
        """
        Unregisters a name from the registry.

        Parameters:
        -----------
        name : str
            The name to be unregistered.
        """
        cls._names.discard(name)

    def clear(cls):
        """
