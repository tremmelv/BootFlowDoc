    """
    A class to represent the boot flow, managing processors, interfaces, triggers, groups, variables, and software.

    Attributes:
    -----------
    processors : List[Processor]
        A list of processors in the boot flow.
    interfaces : List[Interface]
        A list of interfaces in the boot flow.
    triggers : Dict[str, Trigger]
        A dictionary of triggers keyed by their names.
    groups : Dict[str, Group]
        A dictionary of groups keyed by their names.
    variables : Dict[str, float]
        A dictionary of variables and their values.
    software : Dict[str, Software]
        A dictionary of software keyed by their names.
    main_processor : Union[Processor, None]
        The main processor in the boot flow.
    """

    def __init__(self):
        """

        self.processors: List[Processor] = []
        self.interfaces: List[Interface] = []
        self.triggers: Dict[str, Trigger] = {}
        self.groups: Dict[str, Group] = {}
        self.variables: Dict[str, float] = {}
        self.software: Dict[str, Software] = {}
        self.main_processor: Union[Processor, None] = None

[docs]    def remove_item(self, item, processor_name: Union[str, None] = None):
        """
        Removes an item from the boot flow or a specific processor.

        Parameters:
        -----------
        item : object
            The item to be removed.
        processor_name : Union[str, None], optional
            The name of the processor from which the item should be removed (default is None).
        """
        if isinstance(item, Interface) and item in self.interfaces:
            self.interfaces.remove(item)
        elif processor_name:
            processor = self.get_processor(processor_name)
            processor.remove_item(item)



[docs]    def update_duration(self):
        """
        Updates the duration of events in all processors within the boot flow.
        """
        for processor in self.processors:
            for event in processor.boot_events:
                if isinstance(event, ProgramTransfer) or isinstance(event, SizeOperation):
                    event.duration = event.calculate_duration()

    def __str__(self):
        """
        Returns a string representation of the BootFlow instance.

        Returns:
        --------
        str
            A string representation of the BootFlow instance.
        """
        interfaces_str = ', '.join(str(i) for i in self.interfaces)
        processors_str = ', '.join(str(p) for p in self.processors)
        return f"interfaces=[{interfaces_str}], processors=[{processors_str}], triggers=[{self.triggers.keys()}])"


[docs]    def to_dict(self) -> Dict:
        """
        Converts the BootFlow instance to a dictionary.

        Returns:
        --------
        Dict
            A dictionary representation of the BootFlow instance.
        """
        return {
            "interfaces": [i.to_dict() for i in self.interfaces],
            "processors": [p.to_dict() for p in self.processors],
            "triggers": {name: trigger.to_dict() for name, trigger in self.triggers.items()},
            "groups": {name: group.to_dict() for name, group in self.groups.items()},
            "variables": {name: value for name, value in self.variables.items()},
            "software": {name: software.to_dict() for name, software in self.software.items()}
        }



[docs]    @classmethod
    def from_dict(cls, data):
        """
        Creates a BootFlow instance from a dictionary.

        Parameters:
        -----------
        data : Dict
            The dictionary containing boot flow data.

        Returns:
        --------
        BootFlow
            The BootFlow instance created from the dictionary.
        """
        bootflow = cls()
        NameRegistry.clear()
        for name, value in data["variables"].items():
            bootflow.variables[name] = value
        for interface_data in data["interfaces"]:
            bootflow.add_interface(Interface.from_dict(interface_data))
        for _, software_data in data["software"].items():
            bootflow.add_software(Software.from_dict(software_data, bootflow))
        for group_data in data["groups"]:
            bootflow.add_group(Group.from_dict(group_data))
        for processor_data in data["processors"]:
            bootflow.add_processor(Processor.from_dict(processor_data, bootflow))
        for processor in bootflow.processors:
            for event in processor.boot_events:
                if isinstance(event, TimeEvent):
                    if isinstance(event.wait, str):
                        trigger = bootflow.triggers[event.wait]
                        event.wait = Wait(trigger, event)
        return bootflow
