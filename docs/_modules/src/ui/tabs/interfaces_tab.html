[docs]    def create_widgets(self):
        """
        Initializes and arranges the widgets within the tab, setting up the grid
        configuration, input fields, treeview, and comment field.
        """
        master = self.master
        
        # Configure the grid to expand
        master.grid_rowconfigure(0, weight=1)
        master.grid_rowconfigure(1, weight=1)
        master.grid_rowconfigure(2, weight=1)
        master.grid_rowconfigure(4, weight=5)
        master.grid_columnconfigure(2, weight=2)

        # Create a fixed-size frame for input fields and button
        self.input_frame = CustomInputFrame(
            master, width=430, height=250, title="Add/Update Interface",
            fields=[("Name:", ctk.CTkEntry), ("Transfer Speed:", ctk.CTkEntry)],
            button_text="Add/Update Interface", button_command=self.add_or_update_interface,
            tab=self,
            delete_button_text="Delete Selected Interface", delete_command=self.delete_interface
        )
        self.input_frame.grid(row=0, column=0, columnspan=2, rowspan=4, padx=20, pady=20, sticky="nw")

        # Modern Treeview for displaying interfaces
        self.tree = CustomModernTreeview(master, columns=("Name", "Transfer Speed [MB/s]"), theme=self.app.theme)
        self.tree.grid(row=0, column=2, rowspan=5, padx=20, pady=20, sticky="nsew")

        # Comment field
        self.comment_field = CommentField(master)
        self.comment_field.grid(row=4, column=0, columnspan=2, padx=20, pady=20, sticky="nw")

        # Bind selection event
        self.tree.bind("<<TreeviewSelect>>", self.on_tree_select)
        self.master.bind("<Button-1>", self.clear_selection)
        self.tree.bind("<Button-1>", self.clear_selection)
        self.app.bind_all("<Escape>", self.clear_selection)

[docs]    def add_or_update_interface(self):
        """
        Adds a new interface to the BootFlow or updates an existing one based on
        the input provided in the input fields.
        """
        name = self.input_frame.get_entry("Name:")
        speed = self.input_frame.get_entry("Transfer Speed:")

        if speed:
            speed = float(speed)
            selected_item = self.tree.selection()

            if selected_item:
                # Update the selected interface
                selected_item = selected_item[0]
                item_values = self.tree.item(selected_item, "values")

                for interface in self.boot_flow.interfaces:
                    if interface.name == item_values[0]:
                        # Update the interface
                        interface.name = name
                        interface.transferSpeed = speed
                        break
                
                # Refresh the tree view
                self.update_listbox()
                self.app.transfer_tab_frame.update_listbox()
            else:
                # Add a new interface
                interface = Interface(name, speed)
                self.boot_flow.add_interface(interface)
                self.update_listbox()



[docs]    def update_listbox(self):
        """
        Refreshes the treeview with the current list of interfaces from the BootFlow.
        """
        # Add all the elements from boot_flow to the treeview
        data = [(interface.name, interface.transferSpeed) for interface in self.boot_flow.interfaces]
        self.tree.update_treeview(data)



[docs]    def delete_interface(self):
        """
        Deletes the selected interface from the BootFlow and updates the treeview.
        """
        selected_item = self.tree.selection()
        if not selected_item:
            return
        
        selected_item = selected_item[0]
        item_values = self.tree.item(selected_item, "values")  # Get the values of the selected item
        self.boot_flow.del_interface(item_values[0])
        self.update_listbox()  # Refresh the tree view



[docs]    def on_tree_select(self, event):
        """
        Handles the event when an interface is selected in the treeview, populating
        the input fields and comment box with the selected interface's details.

        Parameters:
        -----------
        event : Event
            The event object associated with the treeview selection action.
        """
        selected_item = self.tree.selection()
        if not selected_item:
            return
        
        selected_item = selected_item[0]
        item_values = self.tree.item(selected_item, "values")
        self.comment_field.set_comment_box(item_values[0], self.boot_flow)

        # Populate the input fields with the selected item's values
        self.input_frame.set_entry("Name:", item_values[0])
        self.input_frame.set_entry("Transfer Speed:", item_values[1])
