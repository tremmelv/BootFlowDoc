[docs]class BaseTab:
    """
    A base class for creating tabs in the application, intended to be subclassed by specific tabs.

    Attributes:
    -----------
    master : ctk.CTkFrame
        The parent frame in which the tab's widgets are placed.
    boot_flow : BootFlow
        An instance of BootFlow that the tab interacts with.
    app : object
        A reference to the main application, allowing interaction with other parts of the app.
    """

    def __init__(self, master: ctk.CTkFrame, boot_flow: BootFlow, app):
        """
        Initializes the BaseTab with the given master frame, BootFlow instance, and application reference.

        Parameters:
        -----------
        master : ctk.CTkFrame
            The parent frame for the tab's widgets.
        boot_flow : BootFlow
            An instance of BootFlow that the tab will use.
        app : object
            A reference to the main application.
        """
        self.master = master
        self.boot_flow = boot_flow
        self.app = app
        self.create_widgets()
        self.master.bind("<Visibility>", self.on_tab_selected)

[docs]    def get_processors(self):
        """
        Retrieves the names of processors from the BootFlow instance.

        Returns:
        --------
        List[str]
            A list of processor names, or ["None"] if no processors are present.
        """
        processors_names = []
        if not self.boot_flow.processors:  # Check if the list is empty
            processors_names.append("None")
            return processors_names
        for processors in self.boot_flow.processors:
            processors_names.append(processors.name)
        return processors_names



[docs]    def update_combobox(self):
        """
        Updates the combobox with the list of processor names retrieved from the BootFlow instance.
        """
        self.input_frame.entries["Processor:"].configure(values=self.get_processors())
