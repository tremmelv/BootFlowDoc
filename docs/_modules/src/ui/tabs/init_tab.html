[docs]    def create_widgets(self):
        """
        Creates and configures the widgets for the tab, including input fields, 
        buttons, comment field, and a treeview for displaying events. 
        It also sets up grid configuration for proper layout management.
        """
        master = self.master
        
        # Configure the grid to expand
        master.grid_rowconfigure(0, weight=1)
        master.grid_rowconfigure(1, weight=1)
        master.grid_rowconfigure(2, weight=1)
        master.grid_rowconfigure(4, weight=5)
        master.grid_columnconfigure(2, weight=2)

        fields = [("Name:", ctk.CTkEntry), ("Duration:", ctk.CTkEntry), ("Processor:", ctk.CTkComboBox)]
        self.input_frame = CustomInputFrame(
            master,
            width=420, 
            height=300, 
            title="Add/Update Initialisations",
            fields=fields, 
            button_text="Add/Update Event",
            button_command=self.add_or_update_event,
            tab=self,
            delete_button_text="Delete Selected Event", 
            delete_command=self.delete_event
        )
        self.input_frame.grid(row=0, column=0, columnspan=2, rowspan=3, padx=20, pady=20, sticky="nw")

        self.comment_field = CommentField(master)
        self.comment_field.grid(row=4, column=0, columnspan=2, padx=20, pady=20, sticky="nw")

        self.tree = CustomModernTreeview(master, columns=("Name", "Duration [ms]", "Processor", "Trigger/Wait"), theme=self.app.theme)
        self.tree.grid(row=0, column=2, rowspan=5, padx=20, pady=20, sticky="nsew")

        # Bind selection event
        self.tree.bind("<<TreeviewSelect>>", self.on_tree_select)
        self.master.bind("<Button-1>", self.clear_selection)
        self.tree.bind("<Button-1>", self.clear_selection)
        self.app.bind_all("<Escape>", self.clear_selection)

[docs]    def add_or_update_event(self):
        """
        Adds a new event or updates an existing one in the BootFlow instance 
        based on the current values in the input fields. The method handles 
        different event types (None, Trigger, Wait) and updates the corresponding
        processor and event attributes.
        """
        name = self.input_frame.get_entry("Name:")
        processor_name = self.input_frame.entries["Processor:"].get()
        selected_radio = self.input_frame.radio_var.get()

        selected_processor = self.boot_flow.get_processor(processor_name)

        selected_item = self.tree.selection()
        # if an item is selected in the tree update that item 
        if selected_item:
            selected_item = selected_item[0]
            item_values = self.tree.item(selected_item, "values")
            
            for processor in self.boot_flow.processors:
                for event in processor.boot_events:
                    if isinstance(event, TimeEvent) and (event.name == item_values[0]):
                        event.name = name
                        event.change_processor(selected_processor)

                        if selected_radio == "None":
                            event.trigger = None
                            event.wait = None
                            if name in self.boot_flow.triggers and item_values[3] == "Trigger":
                                del self.boot_flow.triggers[name]
                            duration = self.input_frame.get_entry("Duration:")
                            if duration:
                                event.duration = self.app.safe_str_to_float(duration)
                        elif selected_radio == "Trigger":
                            event.wait = None
                            new_trigger = Trigger(name, event)
                            self.boot_flow.triggers[name] = new_trigger
                            duration = self.input_frame.get_entry("Duration:")
                            if duration:
                                event.duration = self.app.safe_str_to_float(duration)
                        elif selected_radio == "Wait":
                            if name in self.boot_flow.triggers and item_values[3] == "Trigger":
                                event.trigger = None
                                del self.boot_flow.triggers[name]
                            selected_trigger = self.boot_flow.triggers[self.input_frame.entries["Wait for:"].get()]
                            event.set_wait(Wait(selected_trigger, event))
                            event.duration = selected_trigger.calculate_trigger_time()
                        break

            self.update_listbox()
            self.update_combobox()
        # if nothing is slected just creat a new one
        else:

            comment = self.comment_field.textbox.get("0.0", tk.END).strip()
            if selected_radio == "None":
                duration = self.input_frame.get_entry("Duration:")
                if duration:
                    duration = self.app.safe_str_to_float(duration)
                    event = TimeEvent(name, duration, selected_processor, comment)
                    selected_processor.add_event(event)
                    
            elif selected_radio == "Trigger":
                duration = self.input_frame.get_entry("Duration:")
                if duration:
                    duration = self.app.safe_str_to_float(duration)
                    event = TimeEvent(name, duration, selected_processor, comment)
                    selected_processor.add_event(event)
                    new_trigger = Trigger(name, event)
                    self.boot_flow.triggers[name] = new_trigger
                    
            elif selected_radio == "Wait":
                selected_trigger = self.boot_flow.triggers[self.input_frame.entries["Wait for:"].get()]
                duration = selected_trigger.calculate_trigger_time()
                event = TimeEvent(name, duration, selected_processor, comment)
                selected_processor.add_event(event)
                event.set_wait(Wait(selected_trigger, event))
                
            self.update_listbox()
            self.update_combobox()



[docs]    def update_listbox(self):
        """
        Updates the treeview with the current list of events from all processors in the BootFlow instance. 
        The treeview displays the event name, duration, processor name, and whether the event is a trigger or wait event.
        """
        all_events = []
        
        for processor in self.boot_flow.processors:
            for event in processor.boot_events:
                if isinstance(event, TimeEvent) and event.wait and event.wait.trigger and event.wait.trigger.name not in self.boot_flow.triggers:
                    event.wait.trigger = None
            data = [
                (
                    event.name,
                    f"{event.get_duration()} ({event.duration})" if isinstance(event.duration, str) else f"{round(event.get_duration(), 3)}" if event.get_duration() is not None else "-",
                    event.processor.name,
                    "Trigger" if event.trigger else "Wait" if event.wait else "None"
                ) 
                for event in processor.boot_events if isinstance(event, TimeEvent)
            ]
            all_events.extend(data)
        self.tree.update_treeview(all_events)



[docs]    def delete_event(self):
        """
        Deletes the selected event(s) from the BootFlow instance and updates the treeview accordingly.
        """
        selected_items = self.tree.selection()
        if not selected_items:
            return
        
        for selected_item in selected_items:
            item_values = self.tree.item(selected_item, "values")  # Get the values of the selected item
            
            self.boot_flow.del_event_by_name(item_values[0])
        self.update_listbox()



[docs]    def on_tree_select(self, event):
        """
        Handles the selection of an event in the treeview. 
        Populates the input fields with the selected event's details and 
        updates the comment box with the event's comments.
        
        Parameters:
        -----------
        event : Event
            The event object associated with the treeview selection.
        """
        selected_item = self.tree.selection()
        if not selected_item:
            return
        
        selected_item = selected_item[0]
        item_values = self.tree.item(selected_item, "values")
        self.comment_field.set_comment_box(item_values[0], self.boot_flow)

        event = self.boot_flow.get_event_by_name(item_values[0])
        
        radio_value = item_values[3]
        if radio_value == "Trigger":
            self.input_frame.radio_var.set("Trigger")
        elif radio_value == "Wait":
            self.input_frame.radio_var.set("Wait")
        else:
            self.input_frame.radio_var.set("None")

        self.input_frame.update_ui()

        # Populate the input fields with the selected item's values
        self.input_frame.set_entry("Name:", item_values[0])
        if radio_value == "Wait" and event.wait.trigger:
            self.input_frame.entries["Wait for:"].set(event.wait.trigger.name)
        else:
            self.input_frame.set_entry("Duration:", event.duration)
        self.input_frame.entries["Processor:"].set(item_values[2])
