[docs]    def create_widgets(self):
        """
        Initializes and arranges the widgets within the tab, setting up the grid
        configuration, input fields, treeview, and comment field.
        """
        master = self.master

        # Configure the grid to expand
        master.grid_rowconfigure(0, weight=1)
        master.grid_rowconfigure(1, weight=1)
        master.grid_rowconfigure(2, weight=1)
        master.grid_rowconfigure(4, weight=5)
        master.grid_columnconfigure(2, weight=2)

        # Create a fixed-size frame for input fields and buttons
        fields = [("Name:", ctk.CTkEntry), ("Size:", ctk.CTkEntry)]
        self.input_frame = CustomInputFrame(
            master,
            width=420, 
            height=300, 
            title="Add/Update Software",
            fields=fields, button_text="Add/Update Sofware",
            button_command=lambda: self.add_or_update_software(),
            tab=self,
            delete_button_text="Delete Selected Sofware", 
            delete_command=lambda: self.delete_software()
        )
        self.input_frame.grid(row=0, column=0, columnspan=2, rowspan=3, padx=20, pady=20, sticky="nw")

        self.comment_field = CommentField(master)
        self.comment_field.grid(row=4, column=0, columnspan=2, padx=20, pady=20, sticky="nw")

        self.tree = CustomModernTreeview(master, columns=("Name", "Size [MB]"), theme=self.app.theme)
        self.tree.grid(row=0, column=2, rowspan=5, padx=20, pady=20, sticky="nsew")

        self.tree.bind("<<TreeviewSelect>>", self.on_tree_select)
        self.master.bind("<Button-1>", self.clear_selection)
        self.tree.bind("<Button-1>", self.clear_selection)
        self.app.bind_all("<Escape>", self.clear_selection) 



[docs]    def add_or_update_software(self):
        """
        Adds a new software entry to the BootFlow or updates an existing one based
        on the input provided in the input fields.
        """
        name = self.input_frame.get_entry("Name:")
        size = self.input_frame.get_entry("Size:")

        if size:
            selected_item = self.tree.selection()
            if selected_item:
                item_values = self.tree.item(selected_item[0], "values")
                del self.app.boot_flow.software[item_values[0]]
            self.boot_flow.software[name] = Software(name, self.app.safe_str_to_float(size), self.boot_flow)
            self.update_listbox()
        else:
            return



[docs]    def delete_software(self):
        """
        Deletes the selected software entries from the BootFlow and updates the treeview.
        """
        selected_items = self.tree.selection()
        for selected_item in selected_items:
            item_values = self.tree.item(selected_item, "values")
            self.boot_flow.del_software(item_values[0])
        self.update_listbox()

[docs]    def on_tree_select(self, event):
        """
        Handles the event when a software entry is selected in the treeview, populating
        the input fields and comment box with the selected software's details.

        Parameters:
        -----------
        event : Event
            The event object associated with the treeview selection action.
        """
        selected_item = self.tree.selection()
        if not selected_item:
            return
        
        selected_item = selected_item[0]
        item_values = self.tree.item(selected_item, "values")
        self.comment_field.set_comment_box(item_values[0], self.boot_flow)

        software = self.boot_flow.software[item_values[0]]

        # Populate the input fields with the selected item's values
        self.input_frame.set_entry("Name:", item_values[0])
        self.input_frame.set_entry("Size:", software.size)



[docs]    def update_listbox(self):
        """
        Refreshes the treeview with the current list of software from the BootFlow,
        including details such as name and size.
        """
        data = [(name,
                f"{software.get_size()} ({software.size})" if isinstance(software.size, str) else f"{software.get_size()}",
                ) for name, software in self.boot_flow.software.items()]
        self.tree.update_treeview(data)
